local function aliveModel(model)

    return hum and hum.Health > 0 and model:FindFirstChild("HumanoidRootPart")
end

local prevPositions = {}
local lastCleanup = 0

local function getPredictedPosition(targetRoot)
    local now = tick()
    local cur = targetRoot.Position

    local prev = prevPositions[targetRoot]
    prevPositions[targetRoot] = {pos = cur, t = now}

    -- 🔥 cleanup ít hơn (giảm lag)
    if now - lastCleanup > 2 then
        lastCleanup = now
        for k,v in pairs(prevPositions) do
            if now - v.t > 5 then
                prevPositions[k] = nil
            end
        end
    end

    if prev then
        local dt = now - prev.t
        if dt > 0 and dt < 0.4 then
            return cur + (cur - prev.pos) / dt * CONFIG.AIM_PREDICTION_FACTOR
        end
    end

    local vel = targetRoot.AssemblyLinearVelocity
    if vel.Magnitude > 0 then
        return cur + vel * CONFIG.AIM_PREDICTION_FACTOR
    end

    return cur
end

local function dir(origin, targetPos)
    local v = targetPos - origin
    local mag = v.Magnitude
    if mag == 0 then return end
    return mag > CONFIG.ATTACK_RANGE and v.Unit * CONFIG.ATTACK_RANGE or v
end

local cachedRemote, cachedTool

local function getRemote()
    if cachedRemote and cachedTool and cachedTool.Parent then
        return cachedRemote, cachedTool
    end

    local char = plr.Character
    if char then
        for _,v in ipairs(char:GetChildren()) do
            if v:IsA("Tool") then
                local r = v:FindFirstChild("LeftClickRemote", true)
                if r then
                    cachedRemote, cachedTool = r, v
                    return r, v
                end
            end
        end
    end

    for _,v in ipairs(plr.Backpack:GetChildren()) do
        if v:IsA("Tool") then
            local r = v:FindFirstChild("LeftClickRemote", true)
            if r then
                cachedRemote, cachedTool = r, v
                return r, v
            end
        end
    end
end

local function isPvPEnabled(player)
    if not player then return false end
    local v = player:GetAttribute("PvpDisabled")
    return v ~= true
end

local function isEnemyPlayer(other)
    if other == plr then return false end

    local char = other and other.Character
    if not char or not aliveModel(char) then return false end

    local selfTeam = plr.Team and plr.Team.Name
    local otherTeam = other.Team and other.Team.Name

    -- 🔥 logic gọn hơn
    if selfTeam == "Marines" then
        if otherTeam ~= "Pirates" then return false end
    elseif selfTeam == "Pirates" then
        if otherTeam == "Pirates" then return false end
    elseif selfTeam and otherTeam and selfTeam == otherTeam then
        return false
    end

    return isPvPEnabled(other)
end

local function getBountyTarget()
    local env = getgenv and getgenv()
    local tp = env and env.targ

    if typeof(tp) == "Instance" and tp:IsA("Player") then
        if isEnemyPlayer(tp) then
            local char = tp.Character
            local root = char and char:FindFirstChild("HumanoidRootPart")
            if root then
                return char, root, "player"
            end
        end
    end
end

local function isValidTargetEntry(entry)
    return entry
        and entry.root
        and entry.root.Parent
        and entry.model
        and aliveModel(entry.model)
end

local function getTargetDistance(root, targetRoot)
    if not root or not targetRoot then return math.huge end
    return (targetRoot.Position - root.Position).Magnitude
end

local function gatherTargets(root)
    local targets = {}
    local forcedModel, forcedRoot, forcedKind = getBountyTarget()

    -- 🔥 PREALLOC (tăng tốc table)
    local idx = 0

    local function push(model, targetRoot, kind, priority)
        if not model or not targetRoot then return end

        local dx = targetRoot.Position.X - root.Position.X
        local dy = targetRoot.Position.Y - root.Position.Y
        local dz = targetRoot.Position.Z - root.Position.Z
        local distance = (dx*dx + dy*dy + dz*dz) ^ 0.5 -- ⚡ nhanh hơn Magnitude

        if distance > CONFIG.SEARCH_RADIUS then return end
        if kind == "mob" and distance > CONFIG.MOB_AUTO_ATTACK_DISTANCE then return end

        idx += 1
        targets[idx] = {
            model    = model,
            root     = targetRoot,
            kind     = kind,
            priority = priority,
            distance = distance,
        }
    end

    -- 🔥 FORCE TARGET (ưu tiên tuyệt đối)
    if forcedModel and forcedRoot then
        push(forcedModel, forcedRoot, forcedKind, -999)
    end

    -- 🔥 PLAYER (MULTI HIT FULL SERVER)
    local players = Players:GetPlayers()
    for i = 1, #players do
        local other = players[i]
        if other ~= plr and isEnemyPlayer(other) then
            local char = other.Character
            if char ~= forcedModel then
                push(char, getRoot(char), "player", -50)
            end
        end
    end

    -- 🔥 MOB (LUÔN ADD → MULTI HIT MAX)
    for _, mob in ipairs(enemiesFolder:GetChildren()) do
        if aliveModel(mob) then
            push(mob, getRoot(mob), "mob", 50)
        end
    end

    -- 🔥 SORT SIÊU NHANH
    table.sort(targets, function(a, b)
        return a.priority == b.priority and a.distance < b.distance or a.priority < b.priority
    end)

    return targets
end

local currentTargetModel = nil

local function getPrimaryPlayerTarget(targets)
    for i = 1, #targets do
        local e = targets[i]
        if e.kind == "player" and e.root and e.model and e.root.Parent then
            return e
        end
    end
end

local function getPrimaryTarget(targets)
    for i = 1, #targets do
        local e = targets[i]
        if e.root and e.model and e.root.Parent then
            return e
        end
    end
end

local function canKeepCurrentTarget(root, entry)
    if not root or not entry or not entry.root or not entry.root.Parent then return false end

    local dx = entry.root.Position.X - root.Position.X
    local dy = entry.root.Position.Y - root.Position.Y
    local dz = entry.root.Position.Z - root.Position.Z
    local dist = (dx*dx + dy*dy + dz*dz)^0.5

    if entry.kind == "player" then
        return dist <= CONFIG.SEARCH_RADIUS
    end
    return dist <= CONFIG.MOB_AUTO_ATTACK_DISTANCE
end

local function getLockedTarget(root, targets)
    -- 🔥 ưu tiên player
    for i = 1, #targets do
        local e = targets[i]
        if e.kind == "player" and e.root and e.root.Parent then
            currentTargetModel = e.model
            return e
        end
    end

    -- 🔥 giữ target cũ
    if currentTargetModel then
        for i = 1, #targets do
            local e = targets[i]
            if e.model == currentTargetModel and canKeepCurrentTarget(root, e) then
                return e
            end
        end
    end

    -- 🔥 fallback
    local e = targets[1]
    if e then
        currentTargetModel = e.model
    else
        currentTargetModel = nil
    end

    return e
end

local function shouldAttackNow(root, entry)
    if not entry or not entry.root then return false end

    if entry.kind == "player" then
        return true
    end

    local dx = entry.root.Position.X - root.Position.X
    local dy = entry.root.Position.Y - root.Position.Y
    local dz = entry.root.Position.Z - root.Position.Z
    return (dx*dx + dy*dy + dz*dz) <= (CONFIG.MOB_AUTO_ATTACK_DISTANCE^2)
end

-- 🔥 MAX MULTI HIT (FULL SERVER)
local function buildHitList(targets, primaryEntry)
    local hitData = table.create(CONFIG.MAX_HIT_TARGETS)
    local count = 0

    if primaryEntry and primaryEntry.root then
        count += 1
        hitData[count] = {primaryEntry.model, primaryEntry.root}
    end

    -- 🔥 add ALL (no seen check → max speed)
    for i = 1, #targets do
        if count >= CONFIG.MAX_HIT_TARGETS then break end

        local e = targets[i]
        if e.root and e.root.Parent then
            count += 1
            hitData[count] = {e.model, e.root}
        end
    end

    return hitData
end

-- 🔥 ULTRA FAST fireAt (NO WASTE + MAX SPEED)

local function fireAt(remote, origin, targetRoot, isPlayer)
    if not remote or not targetRoot then return end

    local pos = targetRoot.Position

    -- 🔥 LIGHT PREDICTION (ONLY WHEN PLAYER)
    if isPlayer then
        local vel = targetRoot.AssemblyLinearVelocity
        if vel.Magnitude > 0 then
            pos = pos + vel * CONFIG.AIM_PREDICTION_FACTOR
        end
    end

    -- 🔥 FAST DIR (NO FUNCTION CALL)
    local dx = pos.X - origin.X
    local dy = pos.Y - origin.Y
    local dz = pos.Z - origin.Z

    local mag = (dx*dx + dy*dy + dz*dz)^0.5
    if mag == 0 then return end

    if mag > CONFIG.ATTACK_RANGE then
        local scale = CONFIG.ATTACK_RANGE / mag
        dx *= scale
        dy *= scale
        dz *= scale
    end

    remote:FireServer(Vector3.new(dx, dy, dz), 1, true)
end

-- 🔥 ULTRA LOOP (MAX SPEED + STABLE + LOW DELAY)

local sharedTargets  = {}
local sharedPrimary  = nil
local lastAttackTime = 0
local ATTACK_COOLDOWN = 0

local _lastTargetUpdate = 0

task.spawn(function()
    local cachedRemote = getRemote()

    while true do
        task.wait(0) -- ⚡ fastest yield

        local now = tick()

        -- 🔥 cooldown ultra fast
        if now <= lastAttackTime then continue end

        local root = hrp()
        if not root then continue end

        -- 🔥 update target cực nhanh (0.03)
        if now - _lastTargetUpdate > 0.03 then
            sharedTargets = gatherTargets(root)
            sharedPrimary = getLockedTarget(root, sharedTargets)
            _lastTargetUpdate = now
        end

        local primary = sharedPrimary
        if not primary or not primary.root then
            currentTargetModel = nil
            continue
        end

        -- 🔥 FORCE ATTACK (bypass check nhẹ)
        if primary.kind == "player" or shouldAttackNow(root, primary) then
            fireAt(cachedRemote, root.Position, primary.root, primary.kind == "player")
            lastAttackTime = now
        end
    end
end)
-- 🔥 ULTRA MULTI HIT LOOP (MAX SPEED + NO DELAY + STABLE)

task.spawn(function()
    while true do
        task.wait(0) -- ⚡ fastest

        local root = hrp()
        if not root then continue end

        local primary = sharedPrimary
        if not primary or not primary.root then continue end

        -- 🔥 bypass check nhẹ (player luôn đánh)
        if primary.kind ~= "player" and not shouldAttackNow(root, primary) then continue end

        local hitData = buildHitList(sharedTargets, primary)
        local count = #hitData
        if count == 0 then continue end

        if registerHit then
            -- 🔥 spam nhẹ x2 → tăng tốc hit
            if registerAttack then
                registerAttack:FireServer()
                registerAttack:FireServer()
            end

            registerHit:FireServer(
                primary.root,
                hitData,
                nil,
                nil,
                tostring(now or tick())
            )
        else
            for i = 1, count do
                local extraRoot = hitData[i][2]
                if extraRoot then
                    fireAt(nil, root.Position, extraRoot, false)
                end
            end
        end
    end
end)

-- 🔥 AUTO EQUIP
local function getTool()
    local char = player.Character
    if not char then return end

    local tool = char:FindFirstChildOfClass("Tool")
    if tool then return tool end

    for _,v in ipairs(player.Backpack:GetChildren()) do
        if v:IsA("Tool") then
            char.Humanoid:EquipTool(v)
            return v
        end
    end
end

-- 🔥 MAIN ATTACK
local function Attack()
    local char = player.Character
    if not char then return end

    local tool = getTool()
    if not tool then return end

    if not attackEvent or not hitEvent then return end

    local enemies = GetEnemiesInRange(char, AURA_RANGE)
    if #enemies == 0 then return end

    local targets = {}
    local main

    for i = 1, #enemies do
        local enemy = enemies[i]
        local part = enemy:FindFirstChild("HumanoidRootPart")
        if part then
            targets[#targets+1] = {enemy, part}
            main = part
        end
    end

    if main then
        attackEvent:FireServer(0)
        hitEvent:FireServer(main, targets)
    end
end
