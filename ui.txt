    f.Size = UDim2.new(1,-20,1,-100)

    return f
end

local sections = {
    Home = newSection(),
    Player = newSection(),
    Setting = newSection()
}

-- 🔥 SWITCH
local function switch(name)
    for _,v in pairs(sections) do v.Visible = false end
    sections[name].Visible = true
end

homeTab.MouseButton1Click:Connect(function() switch("Home") end)
playerTab.MouseButton1Click:Connect(function() switch("Player") end)
settingTab.MouseButton1Click:Connect(function() switch("Setting") end)

switch("Home")

-- 🔥 FORMAT
local function format(n)
    if n >= 1e6 then return string.format("%.2fM", n/1e6)
    elseif n >= 1e3 then return string.format("%.1fK", n/1e3)
    else return tostring(n) end
end

-- 🔥 LINE
local function line(y)
    local f = Instance.new("TextLabel", sections.Home)
    
    -- 🔥 FIX TRÀN: trừ padding + anchor
    f.Size = UDim2.new(1, -30, 0, 35)
    f.Position = UDim2.new(0, 15, 0, y)

    f.BackgroundTransparency = 1
    f.Font = Enum.Font.GothamSemibold
    f.TextSize = 14
    f.TextColor3 = Color3.fromRGB(255,235,170)
    f.TextXAlignment = Enum.TextXAlignment.Left
    f.TextWrapped = true

    -- 🔥 FIX: không vượt ra ngoài
    f.ClipsDescendants = true

    return f
end

local target = line(0)
local uptime = line(35)
local totalUp = line(70)
local bounty = line(105)
local gain = line(140)
local totalGain = line(175)

-- 🔥 DATA
local start = tick()
getgenv().TotalGain = getgenv().TotalGain or 0
getgenv().TotalUptime = getgenv().TotalUptime or 0

-- 🔥 LOOP
task.spawn(function()
    while true do
        task.wait(0.5)

        local t = getgenv().Targets
        if t and #t > 0 then
            target.Text = "Target: "..t[1].Name
        else
            target.Text = "Target: Searching..."
        end

        local up = math.floor(tick()-start)
        uptime.Text = "Uptime: "..up.."s"

        getgenv().TotalUptime += 1
        totalUp.Text = "Total Uptime: "..format(getgenv().TotalUptime).."s"

        local b = 0
        pcall(function()
            local stats = lp.leaderstats
            local val = stats and (stats:FindFirstChild("Bounty") or stats:FindFirstChild("Bounty/Honor"))
            if val then b = val.Value end
        end)

        bounty.Text = "Bounty: "..format(b)
        gain.Text = "Gain: +"..format(getgenv().LastGain or 0)

        getgenv().TotalGain += (getgenv().LastGain or 0)
        totalGain.Text = "Total Gain: +"..format(getgenv().TotalGain)
    end
end)

-- 🔥 DRAG
local dragging, dragStart, startPos
main.InputBegan:Connect(function(i)
    if i.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = i.Position
        startPos = main.Position
    end
end)

main.InputEnded:Connect(function()
    dragging = false
end)

UIS.InputChanged:Connect(function(i)
    if dragging then
        local d = i.Position - dragStart
        main.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + d.X,
            startPos.Y.Scale,
            startPos.Y.Offset + d.Y
        )
    end
end)
