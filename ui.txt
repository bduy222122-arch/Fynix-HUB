-- 🔥 DRAG (FIX CHUẨN)
local dragging, dragStart, startPos

main.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 
    or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = main.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

UIS.InputChanged:Connect(function(input)
    if dragging then
        local delta = input.Position - dragStart
        main.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)

local sections = {

-- 🔥 SWITCH
local function switch(name)
    for _,v in pairs(sections) do v.Visible = false end
    sections[name].Visible = true
end

playerTab.MouseButton1Click:Connect(function() switch("Player") end)
settingTab.MouseButton1Click:Connect(function() switch("Setting") end)

-- 🔥 FORMAT
local function format(n)
    if n >= 1e6 then return string.format("%.2fM", n/1e6)
    elseif n >= 1e3 then return string.format("%.1fK", n/1e3)
    else return tostring(n) end
end

-- 🔥 DATA
local start = tick()
getgenv().TotalGain = getgenv().TotalGain or 0
getgenv().TotalUptime = getgenv().TotalUptime or 0

-- 🔥 LOOP
task.spawn(function()
    while true do
        task.wait(0.5)

        local t = getgenv().Targets
        if t and #t > 0 then
            target.Text = "Target: "..t[1].Name
        else
            target.Text = "Target: Searching..."
        end

        local up = math.floor(tick()-start)
        uptime.Text = "Uptime: "..up.."s"

        getgenv().TotalUptime += 1
        totalUp.Text = "Total Uptime: "..format(getgenv().TotalUptime).."s"

        local b = 0
        pcall(function()
            local stats = lp.leaderstats
            local val = stats and (stats:FindFirstChild("Bounty") or stats:FindFirstChild("Bounty/Honor"))
            if val then b = val.Value end
        end)

        bounty.Text = "Bounty: "..format(b)
        gain.Text = "Gain: +"..format(getgenv().LastGain or 0)

        getgenv().TotalGain += (getgenv().LastGain or 0)
        totalGain.Text = "Total Gain: +"..format(getgenv().TotalGain)
    end
end)
