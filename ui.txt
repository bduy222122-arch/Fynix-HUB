-- 🔥 FORMAT NUMBER (DẤU PHẨY 1,234,567)
local function format(num)
    num = tonumber(num) or 0
    local str = tostring(math.floor(num))

    -- thêm dấu ,
    while true do
        str, k = str:gsub("^(-?%d+)(%d%d%d)", "%1,%2")
        if k == 0 then break end
    end

    return str
end

-- 🔥 FORMAT TIME (GIỮ NGUYÊN)
local function formatTime(sec)
    sec = math.floor(sec)

    local h = math.floor(sec / 3600)
    local m = math.floor((sec % 3600) / 60)
    local s = sec % 60

    if h > 0 then
        return string.format("%dh %02dm", h, m)
    elseif m > 0 then
        return string.format("%dm %02ds", m, s)
    else
        return sec.."s"
    end
end

getgenv().TotalGain = savedGain
getgenv().LastGain = 0

-- 🔥 GAIN TRACKER (ADD TRƯỚC LOOP)
local oldBounty = nil

task.spawn(function()
    local stats = lp:WaitForChild("leaderstats",10)
    local val = stats and (stats:FindFirstChild("Bounty") or stats:FindFirstChild("Bounty/Honor"))
    if not val then return end

    oldBounty = val.Value -- ❌ bỏ lần đầu

    val.Changed:Connect(function(newVal)
        if oldBounty then
            local diff = newVal - oldBounty
            if diff > 0 then
                getgenv().LastGain = diff
                getgenv().TotalGain += diff

                -- 🔥 SAVE FILE
                writefile(fileGain, tostring(getgenv().TotalGain))
            end
        end
        oldBounty = newVal
    end)
end)

        else
            target.Text = "Target: Searching..."

        -- 🔥 UPTIME
        local session = math.floor(tick() - start)
        local total = savedUptime + session

        uptime.Text = "Uptime: "..formatTime(session)
        totalUp.Text = "Total Uptime: "..formatTime(total)

        getgenv().TotalUptime = total

        -- 🔥 BOUNTY

    end
end)

-- 🔥 HP COLOR
local function getHPColor(percent)
    if percent > 70 then
        return Color3.fromRGB(0,255,120)
    elseif percent > 30 then
        return Color3.fromRGB(255,200,0)
    else
        return Color3.fromRGB(255,60,60)
    end
end
