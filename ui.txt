local function createTab(name, order)
    local b = Instance.new("TextButton", tabBar)
    b.Size = UDim2.new(1/3,-6,1,0)
    b.Position = UDim2.new((order-1)/3, (order-1)*6, 0, 0)
    b.BackgroundColor3 = Color3.fromRGB(32,22,50)
    b.Text = name
    b.Font = Enum.Font.GothamSemibold
    b.TextSize = 15

    -- 🔥 COLOR THEO TAB
    local colorMap = {
        HOME = Color3.fromRGB(0,255,180),
        PLAYER = Color3.fromRGB(255,180,0),
        SETTING = Color3.fromRGB(170,0,255)
    }

    local mainColor = colorMap[name] or Color3.fromRGB(255,255,255)
    b.TextColor3 = mainColor

    -- 🔥 BO GÓC
    Instance.new("UICorner", b).CornerRadius = UDim.new(0,10)

    -- 🔥 VIỀN BUTTON (gradient tím + vàng nhẹ)
    local stroke = Instance.new("UIStroke", b)
    stroke.Thickness = 1.2

    local gradStroke = Instance.new("UIGradient", stroke)
    gradStroke.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(255,215,0)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(170,0,255))
    }

    -- 🔥 VIỀN CHỮ (text stroke)
    local txtStroke = Instance.new("UIStroke", b)
    txtStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Contextual
    txtStroke.Thickness = 0.8
    txtStroke.Transparency = 0.4
    txtStroke.Color = mainColor

    return b
end

-- 🔥 FUNCTION CHUNG
function Tabs.AddParagraph(text, tab)
    tab = tab or "Home"
    local parentScroll = scrolls[tab]
    local layout = layouts[tab]

    local frame = Instance.new("Frame", parentScroll)
    frame.Size = UDim2.new(1,-4,0,58)
    frame.BackgroundColor3 = Color3.fromRGB(26,16,45)
    frame.ClipsDescendants = true
    Instance.new("UICorner", frame).CornerRadius = UDim.new(0,12)

    local stroke = Instance.new("UIStroke", frame)
    stroke.Thickness = 1.5

    local grad = Instance.new("UIGradient", stroke)
    grad.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(255,215,0)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(170,0,255))
    }

    local lbl = Instance.new("TextLabel", frame)
    lbl.Size = UDim2.new(1,-16,1,-10)
    lbl.Position = UDim2.new(0,8,0,5)
    lbl.BackgroundTransparency = 1
    lbl.Text = text
    lbl.Font = Enum.Font.GothamMedium
    lbl.TextSize = 14
    lbl.TextColor3 = Color3.fromRGB(255,235,170)
    lbl.TextWrapped = true
    lbl.TextXAlignment = Enum.TextXAlignment.Left

    parentScroll.CanvasSize = UDim2.new(0,0,0,layout.AbsoluteContentSize.Y + 4)

    return {
        Set = function(newText)
            lbl.Text = newText
        end
    }
end


-- 🔥 SWITCH
local current
local function switch(name)
    for _,v in pairs(sections) do v.Visible = false end
    sections[name].Visible = true
    current = name
end

-- 🔥 DRAG
-- 🔥 DRAG ONLY WHEN HOLD + RELEASE STOP

local dragging = false
local dragStart, startPos

main.InputBegan:Connect(function(i)
    if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = i.Position
        startPos = main.Position
    end
end)

main.InputEnded:Connect(function(i)
    if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
        dragging = false -- 🔥 chỉ kéo khi giữ
    end
end)

UIS.InputChanged:Connect(function(i)
    if dragging and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
        local delta = i.Position - dragStart
        main.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)
