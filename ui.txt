local function format(n)
    if n >= 1e6 then return string.format("%.2fM", n/1e6)

-- 🔥 FORMAT NUMBER (DẤU PHẨY 1,234,567)
local function formatnumber(num)
    num = tonumber(num) or 0
    local str = tostring(math.floor(num))

    -- thêm dấu ,
    while true do
        str, k = str:gsub("^(-?%d+)(%d%d%d)", "%1,%2")
        if k == 0 then break end
    end

    return str
end

-- 🔥 FORMAT TIME (GIỮ NGUYÊN)
local function formatTime(sec)
    sec = math.floor(sec)

    local h = math.floor(sec / 3600)
    local m = math.floor((sec % 3600) / 60)
    local s = sec % 60

    if h > 0 then
        return string.format("%dh %02dm", h, m)
    elseif m > 0 then
        return string.format("%dm %02ds", m, s)
    else
        return sec.."s"
    end
end

local function updateList()
    -- Remove rows for players who left
    for playerName, rowFrame in pairs(PlayerRows) do
        if not Players:FindFirstChild(playerName) then
            rowFrame:Destroy()
            PlayerRows[playerName] = nil
        end
    end

    -- Add/Update players
    local currentPlayers = Players:GetPlayers()
    for i = 1, #currentPlayers do
        local p = currentPlayers[i]
        if p ~= LocalPlayer then
            local row = PlayerRows[p.Name]
            if not row then
                row = createRow(p)
                PlayerRows[p.Name] = row
            end

            -- Update Stats Safely
            local char = p.Character
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local localHrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")

            if hum then
                local hp = math.floor(hum.Health)
                local maxHp = math.floor(hum.MaxHealth)
                local percent = (hp / maxHp) * 100
                
                row.HealthStatus.Text = hp .. "/" .. maxHp
                
                if percent > 70 then
                    row.HealthStatus.TextColor3 = Theme.Green
                elseif percent > 30 then
                    row.HealthStatus.TextColor3 = Theme.Yellow
                else
                    row.HealthStatus.TextColor3 = Theme.Red
                end
            else
                row.HealthStatus.Text = "Dead"
                row.HealthStatus.TextColor3 = Theme.Red
            end

            if hrp and localHrp then
                local dist = math.floor((hrp.Position - localHrp.Position).Magnitude)
                row.DistanceStatus.Text = dist .. "m"
            else
                row.DistanceStatus.Text = "---"
            end
        end
    end
    
    PlayerScroll.CanvasSize = UDim2.new(0, 0, 0, ListLayout.AbsoluteContentSize.Y)
end

--// Loop Logic (0.3s refresh)
task.spawn(function()
    while task.wait(0.3) do
        local success, err = pcall(updateList)
        if not success then
            warn("UI Update Error: " .. tostring(err))
        end
    end
end)
